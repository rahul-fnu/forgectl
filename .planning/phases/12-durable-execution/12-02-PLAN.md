<objective>
Implement checkpoint capture at execution phase boundaries, a daemon startup recovery routine that detects interrupted runs and releases stale locks, and wire execution locks into the run lifecycle.

Purpose: After a daemon crash, runs stuck in "running" status are discovered and handled, stale locks are cleaned up, and duplicate concurrent runs are prevented. Checkpoints at phase boundaries provide the metadata needed to determine what happened before the crash.
Output: Recovery routine, checkpoint helpers, lock wiring into execution and daemon startup.
</objective>

1. Create `src/durability/checkpoint.ts`:
   ```typescript
   export interface CheckpointState {
     phase: "prepare" | "execute" | "validate" | "output";
     timestamp: string;
     metadata?: Record<string, unknown>;
   }

   export function saveCheckpoint(
     snapshotRepo: SnapshotRepository,
     runId: string,
     phase: CheckpointState["phase"],
     metadata?: Record<string, unknown>,
   ): void {
     snapshotRepo.insert({
       runId,
       stepName: `after:${phase}`,
       timestamp: new Date().toISOString(),
       state: { phase, timestamp: new Date().toISOString(), ...metadata } satisfies CheckpointState,
     });
   }

   export function loadLatestCheckpoint(
     snapshotRepo: SnapshotRepository,
     runId: string,
   ): CheckpointState | null {
     const snap = snapshotRepo.latest(runId);
     return snap ? (snap.state as CheckpointState) : null;
   }
   ```

2. Create `src/durability/recovery.ts`:
   ```typescript
   export interface RecoveryResult {
     runId: string;
     action: "marked_interrupted";
     reason: string;
   }

   export function recoverInterruptedRuns(
     runRepo: RunRepository,
     snapshotRepo: SnapshotRepository,
   ): RecoveryResult[] {
     const running = runRepo.findByStatus("running");
     const results: RecoveryResult[] = [];
     for (const run of running) {
       const checkpoint = loadLatestCheckpoint(snapshotRepo, run.id);
       const reason = checkpoint
         ? `Interrupted after ${(checkpoint as CheckpointState).phase} phase. Container likely dead after daemon restart.`
         : "Daemon crashed before any checkpoint was saved";
       runRepo.updateStatus(run.id, {
         status: "interrupted",
         completedAt: new Date().toISOString(),
         error: reason,
       });
       results.push({ runId: run.id, action: "marked_interrupted", reason });
     }
     return results;
   }
   ```

   Note: Per research recommendation, v2.0 marks all interrupted runs as failed rather than attempting container re-creation. Only action is "marked_interrupted".
