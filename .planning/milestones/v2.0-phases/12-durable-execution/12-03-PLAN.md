1. Create `src/durability/pause.ts`:

   ```typescript
   export interface PauseContext {
     reason: string;
     phase: string;
     question?: string;         // The question asked of the human (for display)
     serializedState?: unknown;  // Minimal state needed to resume
   }

   export interface ResumeResult {
     runId: string;
     pauseContext: PauseContext;
     humanInput: string;
   }

   export function pauseRun(
     runRepo: RunRepository,
     runId: string,
     context: PauseContext,
   ): void {
     const run = runRepo.findById(runId);
     if (!run) throw new Error(`Run ${runId} not found`);
     if (run.status !== "running") {
       throw new Error(`Cannot pause run ${runId}: status is ${run.status}, expected running`);
     }
     runRepo.updateStatus(runId, {
       status: "waiting_for_input",
       pauseReason: context.reason,
       pauseContext: context,
     });
   }

   export function resumeRun(
     runRepo: RunRepository,
     runId: string,
     humanInput: string,
   ): ResumeResult {
     const run = runRepo.findById(runId);
     if (!run) throw new Error(`Run ${runId} not found`);
     if (run.status !== "waiting_for_input") {
       throw new Error(`Run ${runId} is ${run.status}, not waiting_for_input`);
     }
     const pauseContext = run.pauseContext as PauseContext;
     runRepo.updateStatus(runId, { status: "running" });
     runRepo.clearPauseContext(runId);
     return { runId, pauseContext, humanInput };
   }
   ```

1. In `src/daemon/routes.ts`, add a new route:
   ```typescript
   app.post("/api/v1/runs/:id/resume", async (req, reply) => {
     const { id } = req.params as { id: string };
     const { input } = req.body as { input: string };
     if (!input || typeof input !== "string") {
       return reply.status(400).send({ error: { code: "BAD_REQUEST", message: "input field is required" } });
     }
     try {
       const result = resumeRun(runRepo, id, input);
       return reply.send({ status: "resumed", runId: result.runId });
     } catch (err) {
       const message = err instanceof Error ? err.message : String(err);
       if (message.includes("not found")) {
         return reply.status(404).send({ error: { code: "NOT_FOUND", message } });
       }
       return reply.status(409).send({ error: { code: "CONFLICT", message } });
     }
   });
   ```

   The route needs access to RunRepository. The `registerRoutes` function already receives a `deps` object -- add `runRepo` to the deps. The `runRepo` is already created in `server.ts` so pass it through.
